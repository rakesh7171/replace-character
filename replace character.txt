#include <iostream>
#include <cstring>
using namespace std;

//#include "solution.h"
void replaceCharacter(char input[], char c1, char c2) {
    // Write your code here
    int i = 0;
    while (input[i] != '\0') {
        if (input[i] == c1) {
            input[i] = c2;
        }
        i++;
    }
}

int main() {
    char input[1000000];
    char c1, c2;
    cin >> input;
    cin >> c1 >> c2;
    replaceCharacter(input, c1, c2);
    cout << input << endl;
}
